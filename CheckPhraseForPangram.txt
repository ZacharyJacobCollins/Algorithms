import java.util.ArrayList;
import java.util.Scanner;

public class Solution {

    public static void main(String[] args) {
    	System.out.println(isPangram());
    }
    
    public static String isPangram() {
	   Scanner input = new Scanner(System.in);
       String entered = input.nextLine().toLowerCase();
       ArrayList<Character> enteredList = new ArrayList<Character>(26);
       //fill list
       for (char c : entered.toCharArray()) {
           enteredList.add(c);
       }
       //check if not pangram
       for (char c : "abcdefghijklmnopqrstuvwxyz".toCharArray()) {
       	if(!(enteredList.contains(c))) {
       		return ("not pangram");
       	}
       }
       input.close();
       return "pangram";
    }
}
